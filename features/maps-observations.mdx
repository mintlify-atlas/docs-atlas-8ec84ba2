## Overview

The Wildlife Watcher app provides interactive maps for visualizing deployment locations and recording wildlife observations. The maps feature integrates with GPS, deployment data, and the observation API.

## Maps Screen

The maps screen displays deployment locations on an interactive Google Maps component:

```typescript src/navigation/screens/Maps.tsx
import MapView, { Region, Marker, PROVIDER_GOOGLE } from 'react-native-maps';
import { useGetDeploymentsQuery } from '../../redux/api/deployments';
import { useAppSelector } from '../../redux';

const Maps = () => {
  const { data: deployments } = useGetDeploymentsQuery();
  const { locationEnabled } = useAppSelector(state => state.locationStatus);
  const mapRef = useRef<MapView>(null);
  
  const [region, setRegion] = useState<Region>({
    latitude: -39.0556,
    longitude: 174.0752,
    latitudeDelta: 0.0922,
    longitudeDelta: 0.0421
  });

  return (
    <MapView
      provider={PROVIDER_GOOGLE}
      ref={mapRef}
      style={styles.map}
      region={region}
      onRegionChange={setRegion}
      showsUserLocation={locationEnabled}
      showsMyLocationButton={false}
    >
      {deployments?.map((deployment) =>
        deployment.latitude && deployment.longitude ? (
          <Marker
            key={deployment.id}
            coordinate={{
              latitude: deployment.latitude,
              longitude: deployment.longitude
            }}
            title={deployment.locationName || `Deployment #${deployment.id.slice(-4)}`}
            description={deployment.deploymentComments}
          />
        ) : null
      )}
    </MapView>
  );
};
```

## Map Controls

### Search Bar

```typescript src/navigation/screens/Maps.tsx
import { Searchbar } from 'react-native-paper';
import { useState } from 'react';

const Maps = () => {
  const [searchQuery, setSearchQuery] = useState('');
  
  const handleSearch = async (query: string) => {
    // Implement geocoding or deployment search
    const results = deployments?.filter(d => 
      d.locationName?.toLowerCase().includes(query.toLowerCase())
    );
    
    if (results?.length) {
      const first = results[0];
      if (first.latitude && first.longitude) {
        mapRef.current?.animateToRegion({
          latitude: first.latitude,
          longitude: first.longitude,
          latitudeDelta: 0.01,
          longitudeDelta: 0.01
        });
      }
    }
  };

  return (
    <View style={styles.searchContainer}>
      <Searchbar
        placeholder="Search location"
        onChangeText={setSearchQuery}
        value={searchQuery}
        onSubmitEditing={() => handleSearch(searchQuery)}
      />
    </View>
  );
};
```

### Creating an Observation

```typescript
import { useCreateObservationMutation } from '../../redux/api/observations';
import { useState } from 'react';
import Geolocation from '@react-native-community/geolocation';

const AddObservation = ({ deploymentId }: { deploymentId: string }) => {
  const [notes, setNotes] = useState('');
  const [location, setLocation] = useState<{ latitude: number; longitude: number }>();
  const [createObservation, { isLoading }] = useCreateObservationMutation();

  const captureLocation = () => {
    Geolocation.getCurrentPosition(
      (position) => {
        setLocation({
          latitude: position.coords.latitude,
          longitude: position.coords.longitude
        });
      },
      (error) => console.error(error),
      { enableHighAccuracy: true }
    );
  };

  const handleSubmit = async () => {
    try {
      await createObservation({
        deploymentId,
        timestamp: new Date().toISOString(),
        latitude: location?.latitude,
        longitude: location?.longitude,
        notes
      }).unwrap();

      Alert.alert('Success', 'Observation recorded');
      navigation.goBack();
    } catch (error) {
      Alert.alert('Error', 'Failed to create observation');
    }
  };

  return (
    <WWScreenView>
      <WWTextInput
        label="Notes"
        value={notes}
        onChangeText={setNotes}
        multiline
        numberOfLines={4}
      />

      <WWButton mode="outlined" onPress={captureLocation}>
        Capture Location
      </WWButton>

      {location && (
        <WWText>
          📍 {location.latitude.toFixed(6)}, {location.longitude.toFixed(6)}
        </WWText>
      )}

      <WWButton
        mode="contained"
        onPress={handleSubmit}
        loading={isLoading}
        disabled={isLoading}
      >
        Save Observation
      </WWButton>
    </WWScreenView>
  );
};
```

## Displaying Observations on Map

```typescript
import { Marker } from 'react-native-maps';
import { useGetObservationsQuery } from '../../redux/api/observations';

const Maps = () => {
  const { data: deployments } = useGetDeploymentsQuery();
  const { data: observations } = useGetObservationsQuery();

  return (
    <MapView ...>
      {/* Deployment markers */}
      {deployments?.map((deployment) => (
        <Marker
          key={`deployment-${deployment.id}`}
          coordinate={{
            latitude: deployment.latitude!,
            longitude: deployment.longitude!
          }}
          title={deployment.locationName}
          pinColor="blue"
        />
      ))}

      {/* Observation markers */}
      {observations?.map((observation) =>
        observation.latitude && observation.longitude ? (
          <Marker
            key={`observation-${observation._id}`}
            coordinate={{
              latitude: observation.latitude,
              longitude: observation.longitude
            }}
            title="Wildlife Observation"
            description={observation.notes}
            pinColor="green"
          />
        ) : null
      )}
    </MapView>
  );
};
```

## Location Permissions

Check and request location permissions:

```typescript src/hooks/useLocationStatus.ts
import { useAppSelector } from '../redux';
import { PermissionsAndroid, Platform } from 'react-native';

const requestLocationPermission = async () => {
  if (Platform.OS === 'android') {
    const granted = await PermissionsAndroid.request(
      PermissionsAndroid.PERMISSIONS.ACCESS_FINE_LOCATION,
      {
        title: 'Location Permission',
        message: 'Wildlife Watcher needs access to your location',
        buttonPositive: 'OK'
      }
    );
    return granted === PermissionsAndroid.RESULTS.GRANTED;
  }
  return true; // iOS handles permissions differently
};

const MyComponent = () => {
  const { locationEnabled } = useAppSelector(state => state.locationStatus);
  
  useEffect(() => {
    if (!locationEnabled) {
      requestLocationPermission();
    }
  }, []);
};
```

## Best Practices

<Card title="Performance" icon="gauge-high">
- Filter markers before rendering to reduce overhead
- Use `useMemo` for computed marker arrays
- Implement marker clustering for dense deployments
</Card>

<Card title="User Experience" icon="user">
- Show loading states while fetching location
- Provide visual feedback for location capture
- Display helpful error messages for permission issues
</Card>

<Card title="Accuracy" icon="bullseye">
- Use `enableHighAccuracy: true` for GPS readings
- Store coordinates with 6+ decimal places
- Display coordinate precision in UI
</Card>

<Card title="Privacy" icon="shield">
- Request location permissions only when needed
- Explain why location access is required
- Allow users to manually enter coordinates if preferred
</Card>
